// @lc code=start
class NumArray {
public:
    NumArray(vector<int>& nums) {
        presum.resize(nums.size() + 1);
        for(int i = 0; i < nums.size(); i++)
        {
            presum[i+1] = presum[i] + nums[i];

        return presum[right+1] - presum[left];

// @lc code=start
class NumMatrix {
public:
    NumMatrix(vector<vector<int>>& matrix) {
        int rows = matrix.size();
        int cols = matrix[0].size();

        presum.resize(rows + 1, vector<int64_t>(cols + 1));

        for(int i = 0; i < rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                presum[i+1][j+1] = presum[i][j+1] + presum[i+1][j] - presum[i][j] + matrix[i][j];
            }
        }
    }

    int sumRegion(int row1, int col1, int row2, int col2) {
        int64_t a = presum[row2 + 1][col2 + 1];
        int64_t b = presum[row1][col1];
        int64_t c = presum[row1][col2 + 1];
        int64_t d = presum[row2 + 1][col1];
        return a - c - d + b;
    }

    vector<vector<int64_t>> presum;
};

/**

 * NumMatrix* obj = new NumMatrix(matrix);
 * int param_1 = obj->sumRegion(row1,col1,row2,col2);

 * [1352] 最后 K 个数的乘积

// @lc code=start
class ProductOfNumbers {
public:
    ProductOfNumbers() {
        premul.push_back(1);
    }

    void add(int num) {
        if (num == 0)
        {
            premul.clear();
            premul.push_back(1);
        }
        else
        {
            premul.push_back(premul.back() * num);
        }
    }

    int getProduct(int k) {
        if (k >= premul.size())
        {
            return 0;
        }
        else
        {
            auto idx = premul.size() - 1 - k;
            return premul.back() / premul[idx];
        }
    }

    vector<int> premul;
};

/**

 * ProductOfNumbers* obj = new ProductOfNumbers();
 * obj->add(num);

## 437-路径总和 III {#437-路径总和-iii}

```C++
/*
 * @lc app=leetcode.cn id=437 lang=cpp
 *
 * [437] 路径总和 III
 */

// @lc code=start
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int pathSum(TreeNode* root, int targetSum) {
        pathSumMap[0] = 1;
        return dfs(root, 0, targetSum);
    }

    int dfs(TreeNode* node, int64_t currSum, int targetSum) {
        if (!node)
        {
            return 0;
        }

        int ans = 0;

        currSum += node->val;
        auto it = pathSumMap.find(currSum - targetSum);
        if (it != pathSumMap.end())
        {
            ans += it->second;
        }
        pathSumMap[currSum]++;
        ans += dfs(node->left, currSum, targetSum);
        ans += dfs(node->right, currSum, targetSum);
        pathSumMap[currSum]--;
        return ans;
    }

    unordered_map<int64_t, int> pathSumMap;
};
// @lc code=end


```

 * [525] 连续数组

// @lc code=start
class Solution {
public:
    int findMaxLength(vector<int>& nums) {
        int ans = 0;
        int sum = 0;
        std::unordered_map<int64_t, int> sumMap;
        sumMap[0] = -1;
        for(int i = 0; i < nums.size(); i++)
        {
            sum += (nums[i] == 1) ? 1 : -1;
            auto it = sumMap.find(sum);
            if (it != sumMap.end())
            {
                auto len = i - it->second;
                ans = max(ans, len);
            }
            else
            {
                sumMap[sum] = i;
            }
        }
        return ans;
    }
};
// @lc code=end

 * [1314] 矩阵区域和

// @lc code=start
class Solution {
public:
    vector<vector<int>> matrixBlockSum(vector<vector<int>>& mat, int k) {
        int rows = mat.size();
        int cols = mat[0].size();

        vector<vector<int>> presum;
        vector<vector<int>> ans;
        presum.resize(rows + 1, vector<int>(cols + 1));
        ans.resize(rows, vector<int>(cols));

        for(int i = 0; i < rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                presum[i+1][j+1] = presum[i][j+1] + presum[i+1][j] - presum[i][j] + mat[i][j];
            }
        }

        for(int i = 0; i < rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                int r1 = max(0, i - k);
                int r2 = min(i + k, rows - 1);
                int c1 = max(0, j - k);
                int c2 = min(j + k, cols - 1);

                ans[i][j] = presum[r2 + 1][c2 + 1] + presum[r1][c1] - presum[r2 + 1][c1] - presum[r1][c2 + 1];
            }
        }
        return ans;
    }


};
// @lc code=end

    int pivotIndex(vector<int>& nums) {
        vector<int> presum(nums.size() + 1);
        for(int i = 0; i < nums.size(); i++)
            presum[i+1] = presum[i] + nums[i];
        for(int i = 0; i < nums.size(); i++)
        {
            int lp = i > 0 ? presum[i] : 0;
            int rp = presum.back() - presum[i+1];
            if (lp == rp)
            {
                return i;
            }
        }
        return -1;
    }

 *
 * [238] 除自身以外数组的乘积
 */

// @lc code=start
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        vector<int64_t> lpremul(nums.size() + 1, 1);
        vector<int64_t> rpremul(nums.size() + 1, 1);
        vector<int> ans(nums.size());
        for(int i = 0; i < nums.size(); i++)
        {
            lpremul[i+1] = lpremul[i] * nums[i];
        }

        for(int i = nums.size() - 1; i >= 0; i--)
        {
            rpremul[i] = rpremul[i+1] * nums[i];
        }

        for(int i = 0; i < nums.size(); i++)
        {
            int lp = lpremul[i];
            int rp = rpremul[i+1];
            ans[i] = lp * rp;
        }
        return ans;
    }
};
// @lc code=end

 * [974] 和可被 K 整除的子数组
 */

// @lc code=start
class Solution {
public:
    int subarraysDivByK(vector<int>& nums, int k) {
        unordered_map<int, int> sumMap;
        sumMap[0] = 1;
        int presum = 0;
        int ans = 0;
        for(int i = 0; i < nums.size(); i++)
        {
            presum += nums[i];
            presum %= k;
            if (presum < 0)
            {
                presum += k;
            }

            auto it = sumMap.find(presum);
            if (it != sumMap.end())
            {
                ans += it->second;
            }
            sumMap[presum]++;
        }
        return ans;
    }
};
// @lc code=end

class Solution {

    bool checkSubarraySum(vector<int>& nums, int k) {
        unordered_map<int, int> sumMap;
        int presum = 0;
        sumMap[0] = -1;
        for(int i = 0; i < nums.size(); i++)
        {
            auto x = nums[i];
            presum += x;
            presum %= k;
            auto it = sumMap.find(presum);
            if (it != sumMap.end())
            {
                if (i - it->second >= 2)
                    return true;
            }
            else
            {
                sumMap[presum] = i;
            }
        }
        return false;
    }
};
